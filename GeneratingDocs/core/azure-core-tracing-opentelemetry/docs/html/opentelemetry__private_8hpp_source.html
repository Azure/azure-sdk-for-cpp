27 class OpenTelemetryAttributeSet final : public Azure::Core::Tracing::_internal::AttributeSet,
28 public opentelemetry::common::KeyValueIterable {
29 std::map<std::string, opentelemetry::common::AttributeValue> m_propertySet;
30
31 template <typename T> void AddAttributeToSet(std::string const& attributeName, T value)
32 {
33 m_propertySet.emplace(
34 std::make_pair(attributeName, opentelemetry::common::AttributeValue(value)));
35 }
36
37 public:
38 void AddAttribute(std::string const& attributeName, int32_t value) override
39 {
40 AddAttributeToSet(attributeName, value);
41 }
42
43 void AddAttribute(std::string const& attributeName, int64_t value) override
44 {
45 AddAttributeToSet(attributeName, value);
46 }
47
48 void AddAttribute(std::string const& attributeName, uint64_t value) override
49 {
50 AddAttributeToSet(attributeName, value);
51 }
52 void AddAttribute(std::string const& attributeName, double value) override
53 {
54 AddAttributeToSet(attributeName, value);
55 }
56
57 void AddAttribute(std::string const& attributeName, std::string const& value) override
58 {
59 AddAttributeToSet<std::string const&>(attributeName, value);
60 }
61 void AddAttribute(std::string const& attributeName, const char* value) override
62 {
63 AddAttributeToSet(attributeName, value);
64 }
65
66 void AddAttribute(std::string const& attributeName, bool value) override
67 {
68 AddAttributeToSet(attributeName, value);
69 }
70
77 bool ForEachKeyValue(
78 opentelemetry::nostd::function_ref<
79 bool(opentelemetry::nostd::string_view, opentelemetry::common::AttributeValue)>
80 callback) const noexcept override
81 {
82 for (auto& value : m_propertySet)
83 {
84 if (!callback(value.first, value.second))
85 {
86 return false;
87 }
88 }
89 return true;
90 }
91
95 size_t size() const noexcept override { return m_propertySet.size(); }
96
97 ~OpenTelemetryAttributeSet() {}
98 };
