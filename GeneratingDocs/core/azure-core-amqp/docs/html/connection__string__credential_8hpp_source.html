54 class ServiceBusSasConnectionStringCredential final : public Credentials::TokenCredential {
55 public:
66 ServiceBusSasConnectionStringCredential(
67 const std::string& connectionString,
68 const std::string& entityPath = {})
69 : TokenCredential("ServiceBusSasConnectionStringCredential"),
70 m_connectionParser(connectionString)
71 {
72 // If we weren't able to determine the entity path from the ConnectionStringCredential
73 // constructor, use the entity path passed in by the user.
74 if (m_connectionParser.GetEntityPath().empty())
75 {
76 m_entityPath = entityPath;
77 }
78 else if (!entityPath.empty() && m_connectionParser.GetEntityPath() != entityPath)
79 {
80 // If the user provided an entity path, but it doesn't match the one in the connection
81 // string, throw.
82 throw std::invalid_argument(
83 "Entity Path provided: '" + entityPath
84 + "' does not match connection string entity path: '"
85 + m_connectionParser.GetEntityPath() + "'.");
86 }
87 }
